import { useState } from "react";

export const Counter = () => {
	const [count, setCount] = useState(0);
    const decrement = () => {
		if (count > 1) {
			setCount(count - 1);
		}
	};
    const increment = () => {
		setCount(count + 1);
	};
    const price = getPrice(count);
	const renewal = (price * 0.7).toFixed(2);
	return (
		<div>
			<div>
				<button onClick={decrement} disabled={count <= 1}>-</button>
				<p>Seats: {count}</p>
				<button onClick={increment}>+</button>
			</div>
			<p>Yearly License Price: ${price.toFixed(2)}</p>
			<p>Renewal Price: ${renewal} / year</p>
		</div>
	);
};
