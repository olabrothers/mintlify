import { useState } from "react";

export const Counter = () => {
	const [count, setCount] = useState(0);
	return (
		<div>
			<div>
				<button onClick={() => setCount(count - 1)}>-</button>
				<p>Seats: {count}</p>
				<button onClick={() => setCount(count + 1)}>+</button>
			</div>
			<p>Yearly License Price: ${count.toFixed(2)}</p>
			<p>Renewal Price: ${count.toFixed(2)} / year</p>
		</div>
	);
};
